## Funcoes do JavaScript
```js
Array.prototype.forEach = function(funcao) {
    const array = this
    for(item of array) {
        funcao(item)
    }
}
// undefined

Array.prototype.map = function(funcao) {
    const array = this
    const novoArray = []
    for(item of array) {
        novoArray.push(funcao(item))
    }
    return novoArray
} 
```
